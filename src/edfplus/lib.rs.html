1//! # EDF+ Library for Rust
2//! 
3//! A pure Rust library for reading and writing EDF+ (European Data Format Plus) files.
4//! This library focuses specifically on EDF+ format and provides a safe, efficient API.
5//!
6//! ## Quick Start
7//!
8//! ### Reading an EDF+ file
9//!
10//! ```rust
11//! use edfplus::{EdfReader, EdfWriter, SignalParam, Result};
12//! # use std::fs;
13//!
14//! fn main() -> Result<()> {
15//!     # // Create a test file first
16//!     # let mut writer = EdfWriter::create("test_data.edf")?;
17//!     # writer.set_patient_info("P001", "M", "01-JAN-1990", "Test Patient")?;
18//!     # let signal = SignalParam {
19//!     #     label: "EEG".to_string(),
20//!     #     samples_in_file: 0,
21//!     #     physical_max: 100.0,
22//!     #     physical_min: -100.0,
23//!     #     digital_max: 32767,
24//!     #     digital_min: -32768,
25//!     #     samples_per_record: 256,
26//!     #     physical_dimension: "uV".to_string(),
27//!     #     prefilter: "HP:0.1Hz".to_string(),
28//!     #     transducer: "AgAgCl".to_string(),
29//!     # };
30//!     # writer.add_signal(signal)?;
31//!     # let samples = vec![10.0; 256];
32//!     # for _ in 0..10 { writer.write_samples(&[samples.clone()])?; }
33//!     # writer.finalize()?;
34//! 
35//!     // Open an EDF+ file
36//!     let mut reader = EdfReader::open("test_data.edf")?;
37//!     
38//!     // Get file information
39//!     let header = reader.header();
40//!     println!("Number of signals: {}", header.signals.len());
41//!     println!("File duration: {:.2} seconds", 
42//!         header.file_duration as f64 / 10_000_000.0);
43//!     
44//!     // Read first 1000 samples from signal 0
45//!     if !header.signals.is_empty() {
46//!         let samples = reader.read_physical_samples(0, 1000)?;
47//!         println!("Read {} samples", samples.len());
48//!     }
49//!     
50//!     # // Cleanup
51//!     # drop(reader);
52//!     # fs::remove_file("test_data.edf").ok();
53//!     Ok(())
54//! }
55//! ```
56//!
57//! ### Creating an EDF+ file
58//!
59//! ```rust
60//! use edfplus::{EdfWriter, SignalParam, Result};
61//! # use std::fs;
62//!
63//! fn main() -> Result<()> {
64//!     // Create a writer
65//!     let mut writer = EdfWriter::create("test_output.edf")?;
66//!     
67//!     // Set patient information
68//!     writer.set_patient_info("P001", "M", "01-JAN-1990", "Patient Name")?;
69//!     
70//!     // Define signal parameters
71//!     let signal = SignalParam {
72//!         label: "EEG Fp1".to_string(),
73//!         samples_in_file: 0,  // Will be calculated automatically
74//!         physical_max: 200.0,
75//!         physical_min: -200.0,
76//!         digital_max: 32767,
77//!         digital_min: -32768,
78//!         samples_per_record: 256,  // Sample rate
79//!         physical_dimension: "uV".to_string(),
80//!         prefilter: "HP:0.1Hz LP:70Hz".to_string(),
81//!         transducer: "AgAgCl cup electrodes".to_string(),
82//!     };
83//!     
84//!     // Add the signal
85//!     writer.add_signal(signal)?;
86//!     
87//!     // Generate and write data
88//!     let mut samples = Vec::new();
89//!     for i in 0..256 {
90//!         let t = i as f64 / 256.0;
91//!         let value = 50.0 * (2.0 * std::f64::consts::PI * 10.0 * t).sin();
92//!         samples.push(value);
93//!     }
94//!     
95//!     writer.write_samples(&[samples])?;
96//!     writer.finalize()?;
97//!     
98//!     # // Cleanup
99//!     # fs::remove_file("test_output.edf").ok();
100//!     Ok(())
101//! }
102//! ```
103//! ### Adds an annotation/event to the EDF+ file
104//! 
105//! **⚠️ CRITICAL TIMING CONSTRAINT**
106//! 
107//! Annotations are only saved when their onset time falls within **future data records**.
108//! Once a data record is written with `write_samples()`, no new annotations can be 
109//! added to that time period.
110//! 
111//! **Timing Rules:**
112//! - Add annotations **BEFORE** writing the data records that cover their time range
113//! - Annotations with `onset_seconds` in already-written time periods will be **silently lost**
114//! - This is due to the sequential write architecture - no backtracking is possible
115//! 
116//! ##### Arguments
117//! 
118//! * `onset_seconds` - Time when the event occurred (seconds since recording start)
119//! * `duration_seconds` - Duration of the event in seconds (None for instantaneous events)  
120//! * `description` - UTF-8 text describing the event (max 40 chars effective)
121//! 
122//! ##### Description Length Limit
123//! 
124//! **Warning**: Annotation descriptions are subject to EDF+ format constraints:
125//! - Maximum effective length is **40 characters** in the final TAL (Time-stamped Annotations Lists) data
126//! - Longer descriptions will be **automatically truncated** during file writing
127//! - UTF-8 multi-byte characters may be truncated at byte boundaries, potentially corrupting the text
128//! - This limit is enforced by the EDF+ standard and matches edflib behavior
129//! 
130//! ```rust
131//! // Write 5 seconds of data (5 records)
132//! # use edfplus::{EdfWriter, SignalParam, Result};
133//! # fn main() -> Result<()> {
134//! let mut writer = EdfWriter::create("annotations.edf")?;
135//! # let signal = SignalParam {
136//! #     label: "EEG".to_string(),
137//! #     samples_in_file: 0,
138//! #     physical_max: 100.0,
139//! #     physical_min: -100.0,
140//! #     digital_max: 32767,
141//! #     digital_min: -32768,
142//! #     samples_per_record: 256,
143//! #     physical_dimension: "uV".to_string(),
144//! #     prefilter: "".to_string(),
145//! #     transducer: "".to_string(),
146//! # };
147//! # writer.add_signal(signal)?;
148//! 
149//! // ✅ Good - within file duration [0.0, 5.0)
150//! writer.add_annotation(2.5, None, "Valid event")?;
151//! writer.add_annotation(4.999, None, "Last moment")?;
152//! 
153//! // ❌ Lost - outside file duration
154//! writer.add_annotation(5.0, None, "Will be discarded")?;
155//! writer.add_annotation(6.0, None, "Also discarded")?;
156//! 
157//! for i in 0..5 {
158//!     let samples = vec![0.0; 256];
159//!     writer.write_samples(&[samples])?;
160//! }
161//! # std::fs::remove_file("annotations.edf").ok();
162//! # Ok(())
163//! # }
164//! ```
165//! 
166//! ## Working with Signal Data
167//!
168//! ### Physical vs Digital Values
169//!
170//! EDF+ stores data as 16-bit integers but represents real-world measurements.
171//! The library automatically handles conversion between digital and physical values:
172//!
173//! ```rust
174//! use edfplus::SignalParam;
175//!
176//! let signal = SignalParam {
177//!     label: "Test Signal".to_string(),
178//!     samples_in_file: 1000,
179//!     physical_max: 100.0,   // +100 µV
180//!     physical_min: -100.0,  // -100 µV  
181//!     digital_max: 32767,    // +32767 (16-bit max)
182//!     digital_min: -32768,   // -32768 (16-bit min)
183//!     samples_per_record: 256,
184//!     physical_dimension: "uV".to_string(),
185//!     prefilter: "".to_string(),
186//!     transducer: "".to_string(),
187//! };
188//!
189//! // Convert digital to physical
190//! let digital_value = 16384;  // Half of max digital value
191//! let physical_value = signal.to_physical(digital_value);
192//! assert!((physical_value - 50.0).abs() < 0.1); // Should be ~50 µV
193//!
194//! // Convert physical to digital  
195//! let physical_input = 25.0;  // 25 µV
196//! let digital_output = signal.to_digital(physical_input);
197//! assert!((digital_output - 8192).abs() <= 1); // Should be ~8192
198//! ```
