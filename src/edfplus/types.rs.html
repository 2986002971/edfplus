3/// Supported EDF file types
4/// 
5/// Currently only EDF+ format is supported as it's the modern standard
6/// with support for annotations and extended metadata.
7/// 
8/// This type is used internally for file format validation.
9#[derive(Debug, Clone, PartialEq)]
10pub enum FileType {
11    /// EDF+ format - European Data Format Plus
12    /// 
13    /// This is the recommended format for new recordings as it supports:
14    /// - Annotations and events
15    /// - Extended patient information  
16    /// - Equipment information
17    /// - Standardized field formats
18    EdfPlus,
19}

21/// Signal parameters and metadata
22/// 
23/// Contains all the information needed to describe a signal in an EDF+ file,
24/// including physical and digital value ranges, labels, and conversion parameters.
25#[derive(Debug, Clone)]
26pub struct SignalParam {
27    /// Signal label/name (e.g., "EEG Fp1", "ECG Lead II")
28    /// 
29    /// # Examples
30    /// 
31    /// Common signal labels:
32    /// - EEG signals: "EEG Fp1", "EEG Fp2", "EEG C3", "EEG C4"
33    /// - ECG signals: "ECG Lead I", "ECG Lead II", "ECG Lead V1"
34    /// - EMG signals: "EMG Left Deltoid", "EMG Right Biceps"
35    pub label: String,
36    
37    /// Total number of samples for this signal in the file
38    pub samples_in_file: i64,
39    
40    /// Maximum physical value (e.g., +200.0 µV)
41    /// 
42    /// This represents the highest real-world measurement value
43    /// that corresponds to the digital maximum value.
44    pub physical_max: f64,
45    
46    /// Minimum physical value (e.g., -200.0 µV) 
47    /// 
48    /// This represents the lowest real-world measurement value
49    /// that corresponds to the digital minimum value.
50    pub physical_min: f64,
51    
52    /// Maximum digital value (typically 32767 for EDF+)
53    /// 
54    /// This is the highest integer value that can be stored
55    /// in the file for this signal.
56    pub digital_max: i32,
57    
58    /// Minimum digital value (typically -32768 for EDF+)
59    /// 
60    /// This is the lowest integer value that can be stored
61    /// in the file for this signal.
62    pub digital_min: i32,
63    
64    /// Number of samples per data record
65    /// 
66    /// For a 1-second data record, this equals the sampling frequency.
67    /// For example, 256 samples per record = 256 Hz sampling rate.
68    pub samples_per_record: i32,
69    
70    /// Physical dimension/unit (e.g., "µV", "mV", "BPM")
71    /// 
72    /// # Examples
73    /// 
74    /// Common units:
75    /// - "uV" or "µV" for EEG signals
76    /// - "mV" for ECG signals  
77    /// - "BPM" for heart rate
78    /// - "%" for oxygen saturation
79    /// - "mmHg" for blood pressure
80    pub physical_dimension: String,
81    
82    /// Prefilter information (e.g., "HP:0.1Hz LP:70Hz")
83    /// 
84    /// Describes any analog or digital filtering applied to the signal
85    /// before digitization or storage.
86    pub prefilter: String,
87    
88    /// Transducer type (e.g., "AgAgCl cup electrodes")
89    /// 
90    /// Describes the sensor or electrode used to acquire the signal.
91    pub transducer: String,
92}

94impl SignalParam {
95    /// Calculate the bit value (resolution) for this signal
96    /// 
97    /// This determines how much each digital unit represents in physical units.
98    /// 
99    /// # Examples
100    /// 
101    /// ```rust
102    /// use edfplus::SignalParam;
103    /// 
104    /// let signal = SignalParam {
105    ///     label: "Test".to_string(),
106    ///     samples_in_file: 1000,
107    ///     physical_max: 100.0,
108    ///     physical_min: -100.0,
109    ///     digital_max: 32767,
110    ///     digital_min: -32768,
111    ///     samples_per_record: 256,
112    ///     physical_dimension: "uV".to_string(),
113    ///     prefilter: "".to_string(),
114    ///     transducer: "".to_string(),
115    /// };
116    /// 
117    /// let bit_value = signal.bit_value();
118    /// // For a ±100µV range over ±32767 digital range:
119    /// // bit_value = 200.0 / 65535 ≈ 0.00305 µV per bit
120    /// assert!((bit_value - 0.00305).abs() < 0.0001);
121    /// ```
122    pub fn bit_value(&self) -> f64 {
123        (self.physical_max - self.physical_min) / 
124        (self.digital_max - self.digital_min) as f64
125    }
126    
127    /// Calculate the offset for digital to physical conversion
128    /// 
129    /// This is used internally for the conversion calculations.
130    /// 
131    /// # Examples
132    /// 
133    /// ```rust
134    /// use edfplus::SignalParam;
135    /// 
136    /// let signal = SignalParam {
137    ///     label: "Test".to_string(),
138    ///     samples_in_file: 1000,
139    ///     physical_max: 200.0,
140    ///     physical_min: -200.0,
141    ///     digital_max: 32767,
142    ///     digital_min: -32768,
143    ///     samples_per_record: 256,
144    ///     physical_dimension: "uV".to_string(),
145    ///     prefilter: "".to_string(),
146    ///     transducer: "".to_string(),
147    /// };
148    /// 
149    /// let offset = signal.offset();
150    /// // The offset should position the conversion correctly  
151    /// assert!(offset.abs() < 1.0); // Should be close to zero for symmetric ranges
152    /// ```
153    pub fn offset(&self) -> f64 {
154        self.physical_max / self.bit_value() - self.digital_max as f64
155    }
156    
157    /// Convert a digital value to its corresponding physical value
158    /// 
159    /// # Arguments
160    /// 
161    /// * `digital_value` - The digital value from the EDF file (typically -32768 to 32767)
162    /// 
163    /// # Returns
164    /// 
165    /// The corresponding physical measurement value with proper units
166    /// 
167    /// # Examples
168    /// 
169    /// ```rust
170    /// use edfplus::SignalParam;
171    /// 
172    /// let signal = SignalParam {
173    ///     label: "EEG Fp1".to_string(),
174    ///     samples_in_file: 1000,
175    ///     physical_max: 200.0,   // +200 µV
176    ///     physical_min: -200.0,  // -200 µV
177    ///     digital_max: 32767,
178    ///     digital_min: -32768,
179    ///     samples_per_record: 256,
180    ///     physical_dimension: "uV".to_string(),
181    ///     prefilter: "".to_string(),
182    ///     transducer: "".to_string(),
183    /// };
184    /// 
185    /// // Test maximum value
186    /// let max_physical = signal.to_physical(32767);
187    /// assert!((max_physical - 200.0).abs() < 0.1);
188    /// 
189    /// // Test minimum value  
190    /// let min_physical = signal.to_physical(-32768);
191    /// assert!((min_physical - (-200.0)).abs() < 0.1);
192    /// 
193    /// // Test zero (should be near middle of range)
194    /// let zero_physical = signal.to_physical(0);
195    /// assert!(zero_physical.abs() < 1.0);
196    /// 
197    /// // Test half-scale positive
198    /// let half_physical = signal.to_physical(16384);
199    /// assert!((half_physical - 100.0).abs() < 1.0);
200    /// ```
201    pub fn to_physical(&self, digital_value: i32) -> f64 {
202        self.bit_value() * (self.offset() + digital_value as f64)
203    }
204    
205    /// Convert a physical value to its corresponding digital value
206    /// 
207    /// # Arguments
208    /// 
209    /// * `physical_value` - The real-world measurement value
210    /// 
211    /// # Returns
212    /// 
213    /// The corresponding digital value that should be stored in the EDF file
214    /// 
215    /// # Examples
216    /// 
217    /// ```rust
218    /// use edfplus::SignalParam;
219    /// 
220    /// let signal = SignalParam {
221    ///     label: "ECG Lead II".to_string(),
222    ///     samples_in_file: 1000,
223    ///     physical_max: 5.0,     // +5 mV
224    ///     physical_min: -5.0,    // -5 mV
225    ///     digital_max: 32767,
226    ///     digital_min: -32768,
227    ///     samples_per_record: 256,
228    ///     physical_dimension: "mV".to_string(),
229    ///     prefilter: "".to_string(),
230    ///     transducer: "".to_string(),
231    /// };
232    /// 
233    /// // Test maximum value
234    /// let max_digital = signal.to_digital(5.0);
235    /// assert!((max_digital - 32767).abs() <= 1);
236    /// 
237    /// // Test minimum value
238    /// let min_digital = signal.to_digital(-5.0);
239    /// assert!((min_digital - (-32768)).abs() <= 1);
240    /// 
241    /// // Test zero
242    /// let zero_digital = signal.to_digital(0.0);
243    /// assert!(zero_digital.abs() <= 1);
244    /// 
245    /// // Test positive value
246    /// let pos_digital = signal.to_digital(2.5);
247    /// assert!((pos_digital - 16384).abs() <= 100);
248    /// ```
249    pub fn to_digital(&self, physical_value: f64) -> i32 {
250        let digital = (physical_value / self.bit_value()) - self.offset();
251        digital.round() as i32
252    }

255/// Annotation or event marker in an EDF+ file
256/// 
257/// Annotations are used to mark events, artifacts, or other points of interest
258/// in the recording timeline.
259/// 
260/// # Examples
261/// 
262/// ```rust
263/// use edfplus::Annotation;
264/// 
265/// // Create an annotation for a seizure event
266/// let seizure_event = Annotation {
267///     onset: 1500000000,  // 150 seconds after start (in 100ns units)
268///     duration: 300000000, // 30 seconds duration (in 100ns units)  
269///     description: "Seizure detected".to_string(),
270/// };
271/// 
272/// // Convert onset to seconds
273/// let onset_seconds = seizure_event.onset as f64 / 10_000_000.0;
274/// assert_eq!(onset_seconds, 150.0);
275/// 
276/// // Convert duration to seconds
277/// let duration_seconds = seizure_event.duration as f64 / 10_000_000.0;
278/// assert_eq!(duration_seconds, 30.0);
279/// ```
280#[derive(Debug, Clone)]
281pub struct Annotation {
282    /// Onset time in 100-nanosecond units since recording start
283    /// 
284    /// To convert to seconds: `onset as f64 / 10_000_000.0`
285    pub onset: i64,
286    
287    /// Duration in 100-nanosecond units (-1 if unknown/instantaneous)
288    /// 
289    /// To convert to seconds: `duration as f64 / 10_000_000.0`
290    pub duration: i64,
291    
292    /// UTF-8 description of the event
293    /// 
294    /// Common annotation types:
295    /// - "Sleep stage 1", "Sleep stage 2", "Sleep stage 3", "Sleep stage 4", "Sleep stage REM"
296    /// - "Seizure", "Spike", "Sharp wave"
297    /// - "Movement artifact", "Eye blink", "Muscle artifact"  
298    /// - "Stimulus onset", "Response", "Button press"
299    pub description: String,
300}

303/// 
304/// Contains all metadata about the recording, including patient information,
305/// recording parameters, and signal definitions.
306/// 
307/// # Examples
308/// 
309/// ```rust
310/// use edfplus::{EdfReader, EdfWriter, SignalParam};
311/// # use std::fs;
312/// 
313/// # // Create a test file first
314/// # let mut writer = EdfWriter::create("test_header_example.edf").unwrap();
315/// # writer.set_patient_info("P001", "M", "01-JAN-1990", "Test Patient").unwrap();
316/// # let signal = SignalParam {
317/// #     label: "EEG".to_string(),
318/// #     samples_in_file: 0,
319/// #     physical_max: 100.0,
320/// #     physical_min: -100.0,
321/// #     digital_max: 32767,
322/// #     digital_min: -32768,
323/// #     samples_per_record: 256,
324/// #     physical_dimension: "uV".to_string(),
325/// #     prefilter: "HP:0.1Hz".to_string(),
326/// #     transducer: "AgAgCl".to_string(),
327/// # };
328/// # writer.add_signal(signal).unwrap();
329/// # let samples = vec![10.0; 256];
330/// # writer.write_samples(&[samples]).unwrap();
331/// # writer.finalize().unwrap();
332/// 
333/// let mut reader = EdfReader::open("test_header_example.edf").unwrap();
334/// let header = reader.header();
335/// 
336/// println!("Recording duration: {:.2} seconds", 
337///     header.file_duration as f64 / 10_000_000.0);
338/// println!("Number of signals: {}", header.signals.len());
339/// println!("Patient: {} ({})", header.patient_name, header.patient_code);
340/// println!("Equipment: {}", header.equipment);
341/// 
342/// for (i, signal) in header.signals.iter().enumerate() {
343///     println!("Signal {}: {} ({} {})", 
344///         i, signal.label, signal.physical_dimension, 
345///         signal.samples_per_record);
346/// }
347/// 
348/// # // Cleanup
349/// # drop(reader);
350/// # fs::remove_file("test_header_example.edf").ok();
351/// ```
352pub struct EdfHeader {
353    /// List of all signals in the file (excluding annotation signals)
354    /// 
355    /// Each signal contains its own parameters like sampling rate,
356    /// physical ranges, labels, etc.

358    
359    /// Total duration of the recording in 100-nanosecond units
360    /// 
361    /// To convert to seconds: `file_duration as f64 / 10_000_000.0`
362    /// 
363    /// # Examples
364    /// 
365    /// ```rust
366    /// use edfplus::{EdfReader, EdfWriter, SignalParam};
367    /// # use std::fs;
368    /// 
369    /// # // Create a test file first

372    /// # let signal = SignalParam {
373    /// #     label: "EEG".to_string(),
374    /// #     samples_in_file: 0,
375    /// #     physical_max: 100.0,
376    /// #     physical_min: -100.0,
377    /// #     digital_max: 32767,
378    /// #     digital_min: -32768,
379    /// #     samples_per_record: 256,
380    /// #     physical_dimension: "uV".to_string(),
381    /// #     prefilter: "HP:0.1Hz".to_string(),
382    /// #     transducer: "AgAgCl".to_string(),
383    /// # };
384    /// # writer.add_signal(signal).unwrap();
385    /// # let samples = vec![10.0; 256];
386    /// # writer.write_samples(&[samples]).unwrap();
387    /// # writer.finalize().unwrap();
388    /// 
389    /// let mut reader = EdfReader::open("test_duration.edf").unwrap();
390    /// let header = reader.header();
391    /// 
392    /// let duration_seconds = header.file_duration as f64 / 10_000_000.0;
393    /// let duration_minutes = duration_seconds / 60.0;
394    /// println!("Recording length: {:.1} minutes", duration_minutes);
395    /// 
396    /// # // Cleanup
397    /// # drop(reader);
398    /// # fs::remove_file("test_duration.edf").ok();
399    /// ```
400    pub file_duration: i64,
401    
402    /// Recording start date
403    pub start_date: NaiveDate,
404    
405    /// Recording start time  
406    pub start_time: NaiveTime,
407    
408    /// Subsecond precision for start time (100-nanosecond units)
409    pub starttime_subsecond: i64,
410    
411    /// Number of data records in the file
412    /// 
413    /// Each data record typically represents 1 second of data,

415    pub datarecords_in_file: i64,
416    
417    /// Duration of each data record in 100-nanosecond units
418    ///

421    pub datarecord_duration: i64,
422

427
