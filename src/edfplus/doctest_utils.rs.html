7/// Creates a simple test EDF+ file for documentation examples
8pub fn create_simple_test_file<P: AsRef<Path>>(path: P) -> Result<()> {
9    let mut writer = EdfWriter::create(&path)?;
10    
11    // Set patient info
12    writer.set_patient_info("DOC001", "M", "01-JAN-1990", "Test Patient")?;
13    
14    // Add a simple EEG signal
15    writer.add_signal(SignalParam {
16        label: "EEG Fp1".to_string(),
17        samples_in_file: 0,
18        physical_max: 200.0,
19        physical_min: -200.0,
20        digital_max: 32767,
21        digital_min: -32768,
22        samples_per_record: 256,
23        physical_dimension: "uV".to_string(),
24        prefilter: "HP:0.1Hz LP:70Hz".to_string(),
25        transducer: "AgAgCl cup electrodes".to_string(),
26    })?;
27    
28    // Generate one second of 10Hz sine wave data
29    let mut samples = Vec::new();
30    for i in 0..256 {
31        let t = i as f64 / 256.0;
32        let value = 50.0 * (2.0 * std::f64::consts::PI * 10.0 * t).sin();
33        samples.push(value);
34    }
35    
36    writer.write_samples(&[samples])?;
37    writer.finalize()?;
38    Ok(())
39}

41/// Creates a multi-channel test EDF+ file for documentation examples
42pub fn create_multi_channel_test_file<P: AsRef<Path>>(path: P) -> Result<()> {
43    let mut writer = EdfWriter::create(&path)?;
44    
45    writer.set_patient_info("DOC002", "F", "15-MAR-1985", "Multi Channel Test")?;
46    
47    // Add EEG signal
48    writer.add_signal(SignalParam {
49        label: "EEG C3".to_string(),
50        samples_in_file: 0,
51        physical_max: 200.0,
52        physical_min: -200.0,
53        digital_max: 32767,
54        digital_min: -32768,
55        samples_per_record: 256,
56        physical_dimension: "uV".to_string(),
57        prefilter: "HP:0.1Hz LP:70Hz".to_string(),
58        transducer: "AgAgCl electrodes".to_string(),
59    })?;
60    
61    // Add ECG signal
62    writer.add_signal(SignalParam {
63        label: "ECG Lead II".to_string(),
64        samples_in_file: 0,
65        physical_max: 5.0,
66        physical_min: -5.0,
67        digital_max: 32767,
68        digital_min: -32768,
69        samples_per_record: 256,
70        physical_dimension: "mV".to_string(),
71        prefilter: "HP:0.1Hz LP:100Hz".to_string(),
72        transducer: "Chest electrodes".to_string(),
73    })?;
74    
75    // Generate sample data
76    let mut eeg_samples = Vec::new();
77    let mut ecg_samples = Vec::new();
78    
79    for i in 0..256 {
80        let t = i as f64 / 256.0;
81        
82        // EEG: Alpha wave (10 Hz) with noise
83        let eeg = 30.0 * (2.0 * std::f64::consts::PI * 10.0 * t).sin()
84                + 5.0 * (2.0 * std::f64::consts::PI * 50.0 * t).sin();
85        eeg_samples.push(eeg);
86        
87        // ECG: Heart beat pattern (60 BPM)
88        let ecg = 2.0 * (2.0 * std::f64::consts::PI * 1.0 * t).sin();
89        ecg_samples.push(ecg);
90    }
91    
92    writer.write_samples(&[eeg_samples, ecg_samples])?;
93    writer.finalize()?;
94    Ok(())
95}

97/// Creates a test file with known digital value ranges for validation
98pub fn create_validation_test_file<P: AsRef<Path>>(path: P) -> Result<()> {
99    let mut writer = EdfWriter::create(&path)?;
100    
101    writer.set_patient_info("VAL001", "X", "X", "Validation Test")?;
102    
103    // Add signal with specific ranges for testing
104    writer.add_signal(SignalParam {
105        label: "Test Signal".to_string(),
106        samples_in_file: 0,
107        physical_max: 100.0,
108        physical_min: -100.0,
109        digital_max: 1000,
110        digital_min: -1000,
111        samples_per_record: 10,
112        physical_dimension: "uV".to_string(),
113        prefilter: "None".to_string(),
114        transducer: "Test Sensor".to_string(),
115    })?;
116    
117    // Generate predictable test data
118    let samples = vec![100.0, 50.0, 0.0, -50.0, -100.0, 75.0, 25.0, -25.0, -75.0, 0.0];
119    writer.write_samples(&[samples])?;
120    writer.finalize()?;
121    Ok(())
122}
